-- DROP FUNCTION public.fn_update_entity_network_status(int4, varchar, varchar, int4);

CREATE OR REPLACE FUNCTION public.fn_update_entity_network_status(p_system_id integer, p_entity_type character varying, p_current_status character varying, p_userid integer)
 RETURNS TABLE(status boolean, message character varying)
 LANGUAGE plpgsql
AS $function$

DECLARE 
    sql TEXT;
v_Query text;
v_tablename character varying;
v_Geomtype character varying;
v_Geomtablename character varying;
v_BuildingStatus character varying;
arow record;
v_regex character varying;
	
 begin
	select value into v_regex from global_settings where key='SqlInjectionRegex';
	if (p_entity_type !~* v_regex or p_current_status !~* v_regex) then
		
		RETURN QUERY
		select '' :: boolean as status,'' :: character varying as message;
	else
	begin

	select layer_table,geom_type from  layer_details where  upper(layer_name)=''||upper($2)||''  into v_tablename,v_Geomtype ;

--for building only --
	if(upper(p_entity_type)=upper('Building'))then
	 select upper(building_status) from att_details_building where system_id=p_system_id  into v_BuildingStatus;
	end if;

	if(v_BuildingStatus ='APPROVED')then
	v_Geomtype :='polygon';
	end if;
	
	
	if lower(v_Geomtype)='point' then
	v_Geomtablename:='point_master'; 
	end if;
	if lower(v_Geomtype)='line' then
	v_Geomtablename:='line_master'; 
	end if;
	if lower(v_Geomtype)='polygon' then
	v_Geomtablename:='polygon_master'; 
	end if;

	

	if(p_current_status='D' and (select count(1) from connection_info 
	where ((source_system_id=p_system_id and upper(source_entity_type)=upper(p_entity_type)) 
	or(destination_system_id=p_system_id and upper(destination_entity_type)=upper(p_entity_type))) and (source_system_id!=destination_system_id and upper(source_entity_type)!=upper(destination_entity_type)))>0)
        then
        RETURN QUERY

		select false as status, (p_entity_type||' can not convert into dormant, as it is used in splicing!')::character varying as message;

	elsif(p_current_status='D') and  exists(select 1 from  associate_entity_info asso where ((asso.associated_system_id=p_system_id and upper(asso.associated_entity_type)=UPPER(p_entity_type)) or 
	(asso.entity_system_id=p_system_id and upper(asso.entity_type)=UPPER(p_entity_type))) and asso.is_termination_point=false)
        then
        
		RETURN QUERY select false as status, (p_entity_type||' can not convert into dormant, as it is associated with some entity!')::character varying as message;

        elsif(upper(p_entity_type)='CABLE' and p_current_status='A'
        and (select count(1) from form_input_settings where upper(form_name)=upper('cable')  and upper(form_feature_name)=upper('start_end_reading') and is_active=true)>0             
        and (select count(1) from att_details_cable where system_id=p_system_id and (cable_calculated_length is null or cable_calculated_length=0))>0
        )
        then
		
		RETURN QUERY select false as status, 'Please update the cable calculated length!'::character varying as message;
	
        else
        raise info'v_tablename%',v_tablename;
        raise info'p_current_status%',p_current_status;
        raise info'p_system_id%',p_system_id;

        
	 EXECUTE 'update  '||v_tablename||' set network_status='''||$3||''', modified_by='||p_userid||',modified_on=now()  where system_id='||p_system_id||'';
         EXECUTE 'update  '||v_Geomtablename||' set network_status='''||$3||''', modified_on=now()  where system_id='||p_system_id||' and entity_type='''||$2||'''';
         
	if(upper(p_entity_type)=upper('ADB') or upper(p_entity_type)=upper('BDB') or upper(p_entity_type)=upper('CDB'))
	then	
		update att_details_splitter set network_status=p_current_status,modified_by=p_userid,modified_on=now()
		 where parent_system_id=p_system_id and upper(parent_entity_type)=upper(p_entity_type);
		update point_master set network_status=p_current_status from att_details_splitter where point_master.system_id=att_details_splitter.system_id and upper(entity_type)=upper('splitter')
		and  parent_system_id=p_system_id and upper(parent_entity_type)=upper(p_entity_type);

		update att_details_fms set network_status=p_current_status,modified_by=p_userid,modified_on=now() 
		where parent_system_id=p_system_id and upper(parent_entity_type)=upper(p_entity_type);
		update point_master set network_status=p_current_status from att_details_fms where point_master.system_id=att_details_fms.system_id and upper(entity_type)=upper('FMS')
		and  parent_system_id=p_system_id and upper(parent_entity_type)=upper(p_entity_type);
	elsif(upper(p_entity_type)=upper('structure')) and exists (select 1 from polygon_master where system_id=p_system_id and upper(entity_type)=upper(p_entity_type))
	then
		update polygon_master set network_status=p_current_status  where system_id=p_system_id and upper(entity_type)=upper(p_entity_type);		
	end if;
            

	  perform(fn_geojson_update_entity_attribute(p_system_id,p_entity_type,0,1,false));
      RETURN QUERY select true as status, 'Network entity status updated successfully'::character varying as message;

	end if;
 end;
 end if;
END
$function$
;
