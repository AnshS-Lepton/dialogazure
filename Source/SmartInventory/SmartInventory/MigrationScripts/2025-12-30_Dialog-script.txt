CREATE OR REPLACE FUNCTION public.fn_splicing_get_entity_report(p_pageno integer, p_pagerecord integer, p_sortcolname character varying, p_sorttype character varying, p_system_id integer, p_entity_type character varying)
 RETURNS SETOF json
 LANGUAGE plpgsql
AS $function$
DECLARE
v_arow record;
v_arow_next record;
v_arow_child record;
v_arow_next2 record;
v_is_virtual_entity boolean;
V_IS_MIDDLEWARE_ENTITY boolean;
 sql TEXT;
   StartSNo    INTEGER;    
   EndSNo      INTEGER;
   TotalRecords integer; 
BEGIN

SELECT IS_VIRTUAL_PORT_ALLOWED,IS_MIDDLEWARE_ENTITY
INTO V_IS_VIRTUAL_ENTITY,V_IS_MIDDLEWARE_ENTITY FROM LAYER_DETAILS WHERE UPPER(LAYER_NAME)=UPPER(P_ENTITY_TYPE);


create temp table temp_connections
(
connection_id integer,
source_system_id integer,
source_network_id character varying(100),
source_entity_type character varying(100),
source_entity_sub_type character varying,
source_entity_title character varying(100),
source_port_no integer,
source_display_name character varying,
destination_system_id integer,
destination_network_id character varying(100),
destination_entity_type character varying(100),
destination_entity_sub_type character varying,
destination_entity_title character varying(100),
destination_port_no integer,
destination_display_name character varying,
source_port_display_name character varying,
destination_port_display_name character varying,
source_tube_no integer,
destination_tube_no integer,
source_tray_system_id integer,
destination_tray_system_id integer,
IS_CABLE_A_END BOOLEAN NOT NULL DEFAULT FALSE,
IS_PROCESSED BOOLEAN NOT NULL DEFAULT FALSE
) on commit drop;

create temp table temp_final_connections
(
source_display_name character varying,
source_entity_title character varying(100),
source_port_no character varying,
destination_entity_title character varying(100),
destination_port_no character varying,
destination_display_name character varying,
tray_name character varying,
via_entity_display_name character varying,
via_entity_Type character varying,
via_port_no character varying,
using_entity_type character varying,
using_entity_display_name character varying,
using_port_in character varying,
using_port_out character varying,
source_tube_no character varying,
destination_tube_no character varying
) on commit drop;

insert into temp_connections(connection_id, source_system_id,source_network_id,source_entity_type,
source_entity_sub_type,source_port_no,source_display_name,source_entity_title,
destination_system_id,destination_network_id,destination_entity_type,destination_entity_sub_type,
destination_port_no,destination_display_name,destination_entity_title,source_port_display_name
,destination_port_display_name,source_tray_system_id,destination_tray_system_id,IS_CABLE_A_END)

select connection_id, source_system_id,source_network_id,source_entity_type,source_entity_sub_type,source_port_no,source_display_name,
case when upper(ls.layer_name)='EQUIPMENT' then coalesce(ssubtype.layer_title,con.source_entity_sub_type)
else ls.layer_title end as source_entity_title,
destination_system_id,destination_network_id,destination_entity_type,destination_entity_sub_type,
destination_port_no,destination_display_name,
case when upper(ld.layer_name)='EQUIPMENT' then coalesce(dsubtype.layer_title,con.destination_entity_sub_type)
else ld.layer_title end as destination_entity_title,
fn_get_port_text(con.source_system_id,(case when upper(con.source_entity_type)='EQUIPMENT' then ssubtype.is_virtual_port_allowed
else ls.is_virtual_port_allowed end),con.source_port_no,con.source_entity_type),
fn_get_port_text(con.destination_system_id,(case when upper(con.destination_entity_type)='EQUIPMENT'
then dsubtype.is_virtual_port_allowed else ld.is_virtual_port_allowed end),con.destination_port_no,con.destination_entity_type)
,con.source_tray_system_id,con.destination_tray_system_id,CON.IS_CABLE_A_END
from connection_info con
left join layer_details ls on upper(ls.layer_name)=upper(con.source_entity_type)
left join layer_details ld on upper(ld.layer_name)=upper(con.destination_entity_type)
left join layer_details ssubtype
on upper(con.source_entity_sub_type)=upper(ssubtype.layer_name) and (ssubtype.isvisible=true or ssubtype.is_visible_in_ne_library=true)
and ssubtype.is_middleware_entity=true
left join layer_details dsubtype
on upper(con.destination_entity_sub_type)=upper(dsubtype.layer_name)
and (dsubtype.isvisible=true or dsubtype.is_visible_in_ne_library=true)
and dsubtype.is_middleware_entity=true
where ((upper(source_entity_type)=upper(p_entity_type) and source_system_id=p_system_id)
or (upper(destination_entity_type)=upper(p_entity_type) and destination_system_id=p_system_id));

update temp_connections t set source_tube_no=c.tube_number from att_details_cable_info c
where c.cable_id=t.source_system_id and upper(t.source_entity_type)=upper('CABLE') and t.source_port_no=c.fiber_number;

update temp_connections t set destination_tube_no=c.tube_number from att_details_cable_info c
where c.cable_id=t.destination_system_id and upper(t.destination_entity_type)=upper('CABLE') and t.destination_port_no=c.fiber_number;

IF(V_IS_MIDDLEWARE_ENTITY)
THEN
FOR V_AROW IN SELECT * FROM TEMP_CONNECTIONS WHERE IS_PROCESSED=FALSE
LOOP
	IF UPPER(V_AROW.DESTINATION_ENTITY_TYPE)='CABLE' --AND
	--NOT EXISTS(SELECT 1 FROM ATT_DETAILS_MODEL WHERE UPPER(NETWORK_ID)=UPPER(V_AROW.SOURCE_NETWORK_ID))
	THEN
		INSERT INTO TEMP_FINAL_CONNECTIONS(SOURCE_PORT_NO,SOURCE_DISPLAY_NAME,SOURCE_ENTITY_TITLE,DESTINATION_PORT_NO,
		DESTINATION_DISPLAY_NAME,DESTINATION_ENTITY_TITLE,DESTINATION_TUBE_NO)
		VALUES(V_AROW.SOURCE_PORT_DISPLAY_NAME,V_AROW.SOURCE_DISPLAY_NAME,
		V_AROW.SOURCE_ENTITY_TITLE,V_AROW.DESTINATION_PORT_DISPLAY_NAME,
		V_AROW.DESTINATION_DISPLAY_NAME,V_AROW.DESTINATION_ENTITY_TITLE,
		V_AROW.DESTINATION_TUBE_NO);
		UPDATE TEMP_CONNECTIONS SET IS_PROCESSED=TRUE WHERE CONNECTION_ID=V_AROW.CONNECTION_ID;
	ELSIF UPPER(V_AROW.SOURCE_ENTITY_TYPE)='CABLE' --AND
	--NOT EXISTS(SELECT 1 FROM ATT_DETAILS_MODEL WHERE UPPER(NETWORK_ID)=UPPER(V_AROW.DESTINATION_NETWORK_ID))
	THEN
		INSERT INTO TEMP_FINAL_CONNECTIONS(SOURCE_PORT_NO,SOURCE_DISPLAY_NAME,SOURCE_ENTITY_TITLE,
		DESTINATION_ENTITY_TITLE,DESTINATION_DISPLAY_NAME,DESTINATION_PORT_NO,DESTINATION_TUBE_NO)
		VALUES(V_AROW.DESTINATION_DISPLAY_NAME,V_AROW.DESTINATION_ENTITY_TITLE,
		V_AROW.DESTINATION_PORT_DISPLAY_NAME,
		V_AROW.SOURCE_ENTITY_TITLE,V_AROW.SOURCE_DISPLAY_NAME,V_AROW.SOURCE_PORT_DISPLAY_NAME ,
		V_AROW.SOURCE_TUBE_NO);
		UPDATE TEMP_CONNECTIONS SET IS_PROCESSED=TRUE WHERE CONNECTION_ID=V_AROW.CONNECTION_ID;
	END IF;
	END LOOP;
ELSE

FOR V_AROW IN SELECT * FROM TEMP_CONNECTIONS WHERE IS_PROCESSED=FALSE AND UPPER(SOURCE_ENTITY_TYPE)='CABLE'
or UPPER(SOURCE_ENTITY_TYPE)='PATCHCORD'
LOOP

SELECT * INTO V_AROW_NEXT FROM TEMP_CONNECTIONS WHERE SOURCE_SYSTEM_ID=V_AROW.DESTINATION_SYSTEM_ID
AND UPPER(SOURCE_ENTITY_TYPE)=UPPER(V_AROW.DESTINATION_ENTITY_TYPE) AND SOURCE_PORT_NO=V_AROW.DESTINATION_PORT_NO;

UPDATE TEMP_CONNECTIONS SET IS_PROCESSED=TRUE WHERE CONNECTION_ID=V_AROW.CONNECTION_ID;
UPDATE TEMP_CONNECTIONS SET IS_PROCESSED=TRUE WHERE CONNECTION_ID=V_AROW_NEXT.CONNECTION_ID;

IF(UPPER(V_AROW_NEXT.DESTINATION_ENTITY_TYPE)='CABLE')
THEN
INSERT INTO TEMP_FINAL_CONNECTIONS(SOURCE_PORT_NO,SOURCE_DISPLAY_NAME,SOURCE_ENTITY_TITLE,DESTINATION_PORT_NO,
			DESTINATION_DISPLAY_NAME,DESTINATION_ENTITY_TITLE,TRAY_NAME,VIA_ENTITY_DISPLAY_NAME,
			VIA_ENTITY_TYPE,SOURCE_TUBE_NO,DESTINATION_TUBE_NO)
			VALUES(V_AROW.SOURCE_PORT_DISPLAY_NAME,V_AROW.SOURCE_DISPLAY_NAME,
			V_AROW.SOURCE_ENTITY_TITLE,V_AROW_NEXT.DESTINATION_PORT_DISPLAY_NAME,
			V_AROW_NEXT.DESTINATION_DISPLAY_NAME,V_AROW_NEXT.DESTINATION_ENTITY_TITLE,
			FN_GET_DISPLAY_NAME((CASE WHEN V_AROW.SOURCE_TRAY_SYSTEM_ID>0 THEN V_AROW.SOURCE_TRAY_SYSTEM_ID ELSE V_AROW.DESTINATION_TRAY_SYSTEM_ID END),'SPLICETRAY'),
			V_AROW_NEXT.SOURCE_DISPLAY_NAME,V_AROW_NEXT.SOURCE_ENTITY_TYPE,
			V_AROW.SOURCE_TUBE_NO,V_AROW_NEXT.DESTINATION_TUBE_NO);
ELSIF(UPPER(V_AROW_NEXT.DESTINATION_ENTITY_TYPE) IN ('SPLITTER','ONT','HTB'))
THEN
	FOR V_AROW_CHILD IN SELECT * FROM TEMP_CONNECTIONS
	WHERE
	CASE
	WHEN V_IS_VIRTUAL_ENTITY
	THEN
	SOURCE_SYSTEM_ID=V_AROW_NEXT.DESTINATION_SYSTEM_ID
	AND UPPER(SOURCE_ENTITY_TYPE)=UPPER(V_AROW_NEXT.DESTINATION_ENTITY_TYPE)
	WHEN UPPER(V_AROW_NEXT.DESTINATION_ENTITY_TYPE)='HTB'
	THEN
	SOURCE_SYSTEM_ID=V_AROW_NEXT.DESTINATION_SYSTEM_ID
	AND UPPER(SOURCE_ENTITY_TYPE)=UPPER(V_AROW_NEXT.DESTINATION_ENTITY_TYPE)
	AND SOURCE_PORT_NO=V_AROW_NEXT.DESTINATION_PORT_NO
	ELSE
	SOURCE_SYSTEM_ID=V_AROW.DESTINATION_SYSTEM_ID
	AND UPPER(SOURCE_ENTITY_TYPE)=UPPER(V_AROW.DESTINATION_ENTITY_TYPE)
	AND SOURCE_PORT_NO=V_AROW.DESTINATION_PORT_NO
	END

LOOP
	SELECT * INTO V_AROW_NEXT2 FROM TEMP_CONNECTIONS
	WHERE SOURCE_SYSTEM_ID=V_AROW_CHILD.DESTINATION_SYSTEM_ID
	AND UPPER(SOURCE_ENTITY_TYPE)=UPPER(V_AROW_CHILD.DESTINATION_ENTITY_TYPE)
	AND SOURCE_PORT_NO=V_AROW_CHILD.DESTINATION_PORT_NO;

	IF(UPPER(V_AROW_CHILD.DESTINATION_ENTITY_TYPE)='PATCHCORD')
	THEN

	SELECT CONNECTION_ID, SOURCE_SYSTEM_ID,SOURCE_NETWORK_ID,SOURCE_ENTITY_TYPE,SOURCE_ENTITY_SUB_TYPE,SOURCE_PORT_NO,SOURCE_DISPLAY_NAME,
	LS.LAYER_TITLE AS SOURCE_ENTITY_TITLE,
	DESTINATION_SYSTEM_ID,DESTINATION_NETWORK_ID,DESTINATION_ENTITY_TYPE,DESTINATION_ENTITY_SUB_TYPE,
	DESTINATION_PORT_NO,DESTINATION_DISPLAY_NAME,
	LD.LAYER_TITLE AS DESTINATION_ENTITY_TITLE,
	FN_GET_PORT_TEXT(CON.SOURCE_SYSTEM_ID,LS.IS_VIRTUAL_PORT_ALLOWED,CON.SOURCE_PORT_NO,CON.SOURCE_ENTITY_TYPE) as source_port_display_name,
	FN_GET_PORT_TEXT(CON.DESTINATION_SYSTEM_ID,LD.IS_VIRTUAL_PORT_ALLOWED,CON.DESTINATION_PORT_NO,CON.DESTINATION_ENTITY_TYPE) as destination_port_display_name
	,CON.SOURCE_TRAY_SYSTEM_ID,CON.DESTINATION_TRAY_SYSTEM_ID,CON.IS_CABLE_A_END,null as SOURCE_TUBE_NO,null as DESTINATION_TUBE_NO INTO V_AROW_NEXT2
	FROM CONNECTION_INFO CON
	LEFT JOIN LAYER_DETAILS LS ON UPPER(LS.LAYER_NAME)=UPPER(CON.SOURCE_ENTITY_TYPE)
	LEFT JOIN LAYER_DETAILS LD ON UPPER(LD.LAYER_NAME)=UPPER(CON.DESTINATION_ENTITY_TYPE)
	WHERE UPPER(SOURCE_ENTITY_TYPE)=UPPER(V_AROW_CHILD.DESTINATION_ENTITY_TYPE) AND SOURCE_SYSTEM_ID=V_AROW_CHILD.DESTINATION_SYSTEM_ID
	AND SOURCE_PORT_NO=V_AROW_CHILD.DESTINATION_PORT_NO;
	
	
	END IF;

	UPDATE TEMP_CONNECTIONS SET IS_PROCESSED=TRUE WHERE CONNECTION_ID=V_AROW_CHILD.CONNECTION_ID;
	UPDATE TEMP_CONNECTIONS SET IS_PROCESSED=TRUE WHERE CONNECTION_ID=V_AROW_NEXT2.CONNECTION_ID;

	IF(V_IS_VIRTUAL_ENTITY)
	THEN
	INSERT INTO TEMP_FINAL_CONNECTIONS(SOURCE_PORT_NO,SOURCE_DISPLAY_NAME,
						SOURCE_ENTITY_TITLE,DESTINATION_PORT_NO,
						DESTINATION_DISPLAY_NAME,DESTINATION_ENTITY_TITLE,TRAY_NAME,VIA_ENTITY_DISPLAY_NAME,
						VIA_ENTITY_TYPE,VIA_PORT_NO,USING_ENTITY_TYPE,USING_ENTITY_DISPLAY_NAME,
						USING_PORT_IN,USING_PORT_OUT,SOURCE_TUBE_NO,DESTINATION_TUBE_NO)
						VALUES(V_AROW.SOURCE_PORT_DISPLAY_NAME,V_AROW.SOURCE_DISPLAY_NAME,
						V_AROW.SOURCE_ENTITY_TITLE,V_AROW_NEXT2.DESTINATION_PORT_DISPLAY_NAME,
						V_AROW_NEXT2.DESTINATION_DISPLAY_NAME,
						V_AROW_NEXT2.DESTINATION_ENTITY_TITLE,
						FN_GET_DISPLAY_NAME((CASE WHEN V_AROW.SOURCE_TRAY_SYSTEM_ID>0 THEN V_AROW.SOURCE_TRAY_SYSTEM_ID ELSE V_AROW.DESTINATION_TRAY_SYSTEM_ID END),'SPLICETRAY'),
						V_AROW_NEXT2.SOURCE_DISPLAY_NAME,V_AROW_NEXT2.SOURCE_ENTITY_TYPE,V_AROW_NEXT2.SOURCE_PORT_DISPLAY_NAME,
						V_AROW_NEXT.DESTINATION_ENTITY_TYPE,V_AROW_NEXT.DESTINATION_DISPLAY_NAME,
						V_AROW_NEXT.DESTINATION_PORT_DISPLAY_NAME,
						V_AROW_CHILD.SOURCE_PORT_DISPLAY_NAME,
						V_AROW.SOURCE_TUBE_NO,V_AROW_NEXT2.DESTINATION_TUBE_NO);
	ELSIF(UPPER(V_AROW_NEXT.DESTINATION_ENTITY_TYPE)='HTB')
	THEN
	INSERT INTO TEMP_FINAL_CONNECTIONS(SOURCE_PORT_NO,SOURCE_DISPLAY_NAME,
	SOURCE_ENTITY_TITLE,DESTINATION_PORT_NO,
	DESTINATION_DISPLAY_NAME,DESTINATION_ENTITY_TITLE,TRAY_NAME,VIA_ENTITY_DISPLAY_NAME,
	VIA_ENTITY_TYPE,VIA_PORT_NO,USING_ENTITY_TYPE,USING_ENTITY_DISPLAY_NAME,
	USING_PORT_IN,USING_PORT_OUT,SOURCE_TUBE_NO,DESTINATION_TUBE_NO)
	VALUES(V_AROW.SOURCE_PORT_DISPLAY_NAME,V_AROW.SOURCE_DISPLAY_NAME,
	V_AROW.SOURCE_ENTITY_TITLE,
	CASE WHEN UPPER(V_AROW_CHILD.DESTINATION_ENTITY_TYPE)='CABLE'
	THEN V_AROW_CHILD.DESTINATION_PORT_DISPLAY_NAME ELSE V_AROW_NEXT2.DESTINATION_PORT_DISPLAY_NAME END,
	CASE WHEN UPPER(V_AROW_CHILD.DESTINATION_ENTITY_TYPE)='CABLE'
	THEN V_AROW_CHILD.DESTINATION_DISPLAY_NAME ELSE V_AROW_NEXT2.DESTINATION_DISPLAY_NAME END,
	CASE WHEN UPPER(V_AROW_CHILD.DESTINATION_ENTITY_TYPE)='CABLE'
	THEN V_AROW_CHILD.DESTINATION_ENTITY_TITLE ELSE V_AROW_NEXT2.DESTINATION_ENTITY_TITLE END,
	FN_GET_DISPLAY_NAME(V_AROW.SOURCE_TRAY_SYSTEM_ID,'SPLICETRAY'),
	V_AROW_NEXT2.SOURCE_DISPLAY_NAME,V_AROW_NEXT2.SOURCE_ENTITY_TYPE,V_AROW_NEXT2.SOURCE_PORT_DISPLAY_NAME,
	V_AROW_NEXT.DESTINATION_ENTITY_TYPE,V_AROW_NEXT.DESTINATION_DISPLAY_NAME,
	V_AROW.DESTINATION_PORT_DISPLAY_NAME,
	V_AROW_CHILD.SOURCE_PORT_DISPLAY_NAME,
	V_AROW.SOURCE_TUBE_NO,V_AROW_NEXT2.DESTINATION_TUBE_NO);
	ELSE

	INSERT INTO TEMP_FINAL_CONNECTIONS(
	SOURCE_PORT_NO,SOURCE_DISPLAY_NAME,SOURCE_ENTITY_TITLE,
	DESTINATION_PORT_NO,DESTINATION_DISPLAY_NAME,DESTINATION_ENTITY_TITLE,
	USING_ENTITY_TYPE,USING_ENTITY_DISPLAY_NAME,
	USING_PORT_IN,USING_PORT_OUT,SOURCE_TUBE_NO,DESTINATION_TUBE_NO)
	VALUES(V_AROW.SOURCE_PORT_DISPLAY_NAME,V_AROW.SOURCE_DISPLAY_NAME,V_AROW.SOURCE_ENTITY_TITLE, V_AROW_NEXT2.DESTINATION_PORT_DISPLAY_NAME,
	V_AROW_NEXT2.DESTINATION_DISPLAY_NAME,
	V_AROW_NEXT2.DESTINATION_ENTITY_TITLE,
	V_AROW_CHILD.SOURCE_ENTITY_TYPE,
	V_AROW_CHILD.SOURCE_DISPLAY_NAME,
	V_AROW_CHILD.SOURCE_PORT_DISPLAY_NAME,
	V_AROW_NEXT2.SOURCE_PORT_DISPLAY_NAME,
	V_AROW.SOURCE_TUBE_NO,V_AROW_NEXT2.DESTINATION_TUBE_NO);

	END IF;
	END LOOP;
END IF;
END LOOP;
END IF;
 --RETURN QUERY SELECT ROW_TO_JSON(ROW) FROM(
 sql='SELECT ROW_NUMBER() OVER (ORDER BY VIA.LAYER_ID DESC) AS S_No,
  CON.SOURCE_DISPLAY_NAME AS SOURCE,CON.SOURCE_ENTITY_TITLE AS SOURCE_ENTITY_TYPE,CON.SOURCE_TUBE_NO,CON.SOURCE_PORT_NO,CON.TRAY_NAME,
 CON.VIA_ENTITY_DISPLAY_NAME AS VIA_ENTITY,VIA.LAYER_TITLE AS VIA_ENTITY_TYPE,CON.VIA_PORT_NO,
 CON.USING_ENTITY_DISPLAY_NAME AS USING_ENTITY,USNG.LAYER_TITLE AS USING_ENTITY_TYPE,CON.USING_PORT_IN,CON.USING_PORT_OUT,
 CON.DESTINATION_DISPLAY_NAME AS DESTINATION,CON.DESTINATION_ENTITY_TITLE AS DESTINATION_ENTITY_TYPE,CON.DESTINATION_TUBE_NO,CON.DESTINATION_PORT_NO
 FROM TEMP_FINAL_CONNECTIONS CON
 LEFT JOIN LAYER_DETAILS VIA ON UPPER(CON.VIA_ENTITY_TYPE)=UPPER(VIA.LAYER_NAME)
 LEFT JOIN LAYER_DETAILS USNG ON UPPER(CON.USING_ENTITY_TYPE)=UPPER(USNG.LAYER_NAME) ORDER BY CON.SOURCE_TUBE_NO,CON.SOURCE_PORT_NO';
-- ) ROW;
 EXECUTE   'SELECT COUNT(1)  FROM ('||sql||') as TR' INTO TotalRecords;
 IF((P_PAGENO) <> 0) THEN
	StartSNo:=  P_PAGERECORD * (P_PAGENO - 1 ) + 1;
	EndSNo:= P_PAGENO * P_PAGERECORD;
	sql:= 'SELECT '||TotalRecords||' as totalRecords, *
                FROM (' || sql || ' ) T 
                WHERE S_No BETWEEN ' || StartSNo || ' AND ' || EndSNo; 

 ELSE
         sql:= 'SELECT '||TotalRecords||' as totalRecords, * FROM (' || sql || ' ) T';                  
 END IF; 
RAISE INFO 'queryS_%', sql;
RETURN QUERY
EXECUTE 'select row_to_json(ROW) from ('||sql||') ROW';
END ;
$function$
;

CREATE OR REPLACE FUNCTION public.fn_get_fms_connection_report_new(p_networkstatues character varying, p_provinceids character varying, p_regionids character varying, p_layer_name character varying, p_searchby character varying, p_searchbytext character varying, p_fromdate character varying, p_todate character varying, p_pageno integer, p_pagerecord integer, p_sortcolname character varying, p_sorttype character varying, p_geom character varying, p_duration_based_column character varying, p_radius double precision, p_userid integer)
 RETURNS SETOF json
 LANGUAGE plpgsql
AS $function$
DECLARE
V_AROW RECORD;
v_entity_port_no character varying;
v_maxcount integer;
v_maxpathid integer;
v_ststem_id integer;
v_entity_type integer;
v_maxfms_id integer;
sql TEXT;
StartSNo INTEGER;
EndSNo INTEGER;
LowerStart character varying;
LowerEnd character varying;
TotalRecords integer;
WhereCondition character varying;
s_report_view_name character varying;
s_geom_type character varying;
s_layer_id integer;
s_layer_columns text;
TotalAppliedRecords integer;
TotalRejectedRecords integer;
r RECORD;
ss_report_view_name TEXT;
d_report_view_name TEXT;
v_report_view_name TEXT;
ss_layer_name TEXT;
update_sql TEXT;
s_latitude text;
d_layer_name TEXT;
v_layer_name TEXT;
begin

truncate table cpf_temp_result;
-- drop table cpf_temp_result;
/*
create TEMP table cpf_temp_result
(
id serial,
source_system_id integer,
source_network_id character varying(500),
source_port_no integer,
source_entity_type character varying(100),
source_entity_title character varying(100),
destination_system_id integer,
destination_network_id character varying(500),
destination_port_no integer,
destination_entity_type character varying(100),
destination_entity_title character varying(100),
viya_system_id integer,
via_network_id character varying(500),
path_id integer,
via_entity_type character varying(100),
SOURCE_TUBE_COLOUR_CODE character varying(100),
DESTINATION_TUBE_COLOUR_CODE character varying(100),
SOURCE_PORT_COLOUR_CODE character varying(100),
DESTINATION_PORT_COLOUR_CODE character varying(100),
SOURCE_TUBE_NAME character varying(100),
DESTINATION_TUBE_NAME character varying(100),
fms_id integer,
splitter_id integer,
entity_type character varying(100),
client_id character varying,
client_link_id character varying,
client_core_id character varying,
otdr_length float(8) default (0),
op_alias character varying,
a_site_id character varying,
a_site_name character varying,
b_site_id character varying,
b_site_name character varying
)on commit drop;
*/
-------------------------------------------------------------------------------------------------------------------
SELECT LAYER_ID, REPORT_VIEW_NAME, GEOM_TYPE INTO S_LAYER_ID, S_REPORT_VIEW_NAME,S_GEOM_TYPE FROM LAYER_DETAILS WHERE LOWER(LAYER_NAME) = LOWER(P_LAYER_NAME);
-- SELECT ALL ACTIVE LAYER FIELDS FROM LAYER COLUMN SETTINGS IN DEFINED ORDER..
SELECT STRING_AGG(COLUMN_NAME||' as "'||case when COALESCE(res_field_key,'') ='' then DISPLAY_NAME else res_field_key end||'"', ',') INTO S_LAYER_COLUMNS FROM (
SELECT COLUMN_NAME,res_field_key,DISPLAY_NAME FROM LAYER_COLUMNS_SETTINGS WHERE LAYER_ID=S_LAYER_ID AND UPPER(SETTING_TYPE)='REPORT' AND IS_ACTIVE=TRUE ORDER BY COLUMN_SEQUENCE) A;
IF ((p_geom) != '')
THEN
if(p_radius>0)
then
select substring(left(St_astext(ST_buffer_meters(ST_GeomFromText('POINT('||p_geom||')',4326),p_radius)),-2),10) into p_geom;
end if;
RAISE INFO '------------------------------------S_LAYER_COLUMNS%', S_LAYER_COLUMNS;
sql:= 'SELECT system_id from (select a.system_id from '||S_REPORT_VIEW_NAME||' a
inner join '||s_geom_type||'_master m
on a.system_id=m.system_id and upper(m.entity_type)=upper('''||p_layer_name||''')
and ST_Intersects(m.sp_geometry, st_geomfromtext(''POLYGON(('||p_geom||'))'',4326))
inner join vw_user_permission_area pa on pa.province_id = a.province_id where pa.user_id = '||p_userid||' and 1=1 ';
--inner join vw_user_permission_area pa on pa.province_id = a.province_id where pa.user_id = '||p_userid||' and a.parent_type in(select layer_title from layer_details where layer_name=''POD'') and 1=1 ';
RAISE INFO '------------------------------------sql%', sql;
ELSE
sql:= 'SELECT ROW_NUMBER() OVER (ORDER BY system_id desc) AS S_No,system_id from (select a.system_id from '||S_REPORT_VIEW_NAME||' a
inner join vw_user_permission_area pa on pa.province_id = a.province_id where pa.user_id = '||p_userid||' and 1=1 ';
--inner join vw_user_permission_area pa on pa.province_id = a.province_id where pa.user_id = '||p_userid||' and a.parent_type in(select layer_title from layer_details where layer_name=''POD'') and 1=1 ';
RAISE INFO '------------------------------------sql1%', sql;
END IF;

IF ((p_networkStatues) != '' and upper(S_LAYER_COLUMNS) like '%NETWORK_STATUS%') THEN
sql:= sql ||' AND upper(Cast(a.network_status as TEXT)) in('||p_networkStatues||')';
END IF;
IF ((p_RegionIds) != '') THEN
sql:= sql ||' AND a.region_id IN ('||p_RegionIds||')';
END IF;
IF ((P_ProvinceIds) != '') THEN
sql:= sql ||' AND a.province_id IN ('||P_ProvinceIds||')';

when upper(entity_type)=''TRENCH'' then concat(round((length_qty)::numeric, 2),''(m)/'','||v_total_line||')::character varying
 when upper(entity_type)=''LOOP'' then concat(round((length_qty)::numeric, 2),''(m)/'','||v_total_point_entity||')::character varying else length_qty ::character varying end as length_qty,
(round((length_qty)::numeric, 2) * service_cost_per_unit +  cost_per_unit * round((length_qty)::numeric, 2))as amount,entity_type,geom_type,cost_per_unit,service_cost_per_unit from ('||sql||')e';

 raise info 'inner query fin=al =%',v_outer_query;
  
IF (V_IS_VALID) THEN
    RETURN QUERY
    EXECUTE 'select row_to_json(row) 
             from (select *,'|| quote_literal(coalesce(p_site_id, '')) ||' as site_id 
                   from ('||v_outer_query||')t order by geom_type) row';
ELSE
    RETURN QUERY
    EXECUTE 'select row_to_json(row) 
             from (select '||V_IS_VALID||'::boolean AS is_template_extis, 
                          '||quote_literal(V_MESSAGE)||'::character varying AS msg, 
                          '||quote_literal(coalesce(p_site_id, ''))||'::character varying as site_id) row';
END IF;

            
 END
$function$
