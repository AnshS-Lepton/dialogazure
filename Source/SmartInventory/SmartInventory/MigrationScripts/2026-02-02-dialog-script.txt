	) draftinfo

	ST_Intersects(coalesce(draftinfo.updated_geom,poly.sp_geometry),ST_buffer_meters(St_Geomfromtext('POINT('||lng||' '||lat||')',4326),mtrBuffer)) 
	) tbl1
) tbl join layer_details l on upper(tbl.entity_type)=upper(l.layer_name)

 inner  join role_permission_entity rp on l.layer_id  = rp.layer_id and rp.role_id = v_role_id and rp.network_status = tbl.network_status and rp.viewonly = true
 and l.is_info_enabled=true and case when upper(l.layer_name)='PIT'  then 1=1 else l.isvisible=true end
 
WHERE row_num = 1
and
(
    -- VENDOR USER
    EXISTS (
        SELECT 1
        FROM user_master um
        WHERE um.user_id = p_user_id
          AND lower(um.user_type) = 'vendor'
    )
    AND p_source_ref_id <> ''
    AND
    (
        -- Entity inside polygon
        EXISTS (
            SELECT 1
            FROM polygon_master pm
            WHERE pm.system_id = p_source_ref_id::INTEGER
              AND ST_Within(
                    ST_SetSRID(tbl.geom, 4326),
                    pm.sp_geometry
              )
        )
        OR (
            tbl.entity_type = 'Network_Ticket'   
            AND tbl.system_id = p_source_ref_id::INTEGER
        )
    )
)
OR
(
    NOT EXISTS (
        SELECT 1
        FROM user_master um
        WHERE um.user_id = p_user_id
          AND lower(um.user_type) = 'vendor'
    )
)

 order by tbl.geom_type asc
 , tbl.entity_type desc,tbl.display_name asc;
 --tbl.entity_type;
 Raise info 'Query end to return query ->%', clock_timestamp();
    
END

 SUM(TOTAL_ASBUILT_SERVICE_COST) AS TOTAL_ASBUILT_SERVICE_COST,SUM(TOTAL_DORMANT_SERVICE_COST) AS TOTAL_DORMANT_SERVICE_COST,SUM(TOTAL_PLANNED_COUNT) AS TOTAL_PLANNED_COUNT,
 SUM(TOTAL_ASBUILT_COUNT) AS TOTAL_ASBUILT_COUNT,SUM(TOTAL_DORMANT_COUNT) AS TOTAL_DORMANT_COUNT,SUM(TOTAL_PLANNED_CALC_LENGTH) AS TOTAL_PLANNED_CALC_LENGTH,

 SUM(TOTAL_DORMANT_GIS_LENGTH) AS TOTAL_DORMANT_GIS_LENGTH,TRUE AS IS_HEADER FROM BOM_BOQ_INFO GROUP BY ENTITY_TYPE,GEOM_TYPE;

-- PREAPARE A FINAL QUERY TO RETUNR THE PLANNED, AS-BUILT, DORMANT AND COMBINED BOM/BOQ SEPERATLY IN ROW FORMAT.
V_SqlFinalQuery:='';
V_CommonCost:=''; V_CommonServiceCost :=''; V_CommonCount  :=''; V_CommonCLength  :=''; V_CommonGISLength  :='';

for arow in select regexp_split_to_table(p_network_status, ',') as NetworkStatus
loop
Q_SelectedNetworkStatus:= case when upper(arow.NetworkStatus)='P' then 'PLANNED' when upper(arow.NetworkStatus)='A' then 'ASBUILT'
when upper(arow.NetworkStatus)='D' then 'DORMANT' else '' end ;

if(V_SqlFinalQuery!='') then
V_SqlFinalQuery:=V_SqlFinalQuery||' union All ';
end if;

V_SqlFinalQuery:= V_SqlFinalQuery|| 'SELECT distinct ID, ENTITY_TYPE,ITEM_CODE, GEOM_TYPE, ENTITY_SUB_TYPE, ROUND(COST_PER_UNIT,2)

TOTAL_'||Q_SelectedNetworkStatus||'_COUNT as Total_Count,
ROUND( (TOTAL_'||Q_SelectedNetworkStatus||'_CALC_LENGTH) ,2) as CALCULATED_LENGTH,
ROUND( (TOTAL_'||Q_SelectedNetworkStatus||'_GIS_LENGTH) ,2) as GIS_LENGTH, '''||arow.NetworkStatus||''' as NetworkStatus,
IS_HEADER FROM BOM_BOQ_INFO where TOTAL_'||Q_SelectedNetworkStatus||'_COUNT >0';

-- COLUMUNS FOR COMBNINED SECTION WRITTEN OUTSIDE THE LOOP.
V_CommonCost:=V_CommonCost||case when length(V_CommonCost)>0 then ' + ' else '' end||'ROUND(TOTAL_'||Q_SelectedNetworkStatus||'_COST,2)';
V_CommonServiceCost:=V_CommonServiceCost||case when length(V_CommonServiceCost)>0 then ' + ' else '' end||'ROUND(TOTAL_'||Q_SelectedNetworkStatus||'_SERVICE_COST,2)';

V_CommonGISLength:=V_CommonGISLength||case when length(V_CommonGISLength)>0 then ' + ' else '' end||'ROUND(TOTAL_'||Q_SelectedNetworkStatus||'_GIS_LENGTH,2)';

 end loop;

-- COMBINED BOM/BOQ QUERY..
if( upper(p_network_status) like '%,%') then

V_SqlFinalQuery:= V_SqlFinalQuery|| ' union All
SELECT ID, ENTITY_TYPE, ITEM_CODE, GEOM_TYPE, ENTITY_SUB_TYPE, ROUND(COST_PER_UNIT,2) as COST_PER_UNIT, ROUND(SERVICE_COST_PER_UNIT,2) as SERVICE_COST_PER_UNIT,
 SPECIFICATION, ITEM_CODE, NAME,
('||V_CommonCost||') as TOTAL_COST,
('||V_CommonServiceCost||') as TOTAL_SERVICE_COST,
('||V_CommonCount||') as Total_Count,
round(('||V_CommonCLength||'),2) as CALCULATED_LENGTH,
round(('||V_CommonGISLength||'),2) as GIS_LENGTH, ''C'' as NetworkStatus,
IS_HEADER FROM BOM_BOQ_INFO ';
end if;

raise info 'V_SqlFinalQuery_test----------%',V_SqlFinalQuery;

RETURN QUERY EXECUTE 'select row_to_json(row) from (select * from('||V_SqlFinalQuery||')Fdt order by (case when NetworkStatus=''P'' then 1 when NetworkStatus=''A'' then 2 when NetworkStatus=''D'' then 3 else 4 end), ENTITY_TYPE,IS_HEADER DESC,ID) row';

end;
