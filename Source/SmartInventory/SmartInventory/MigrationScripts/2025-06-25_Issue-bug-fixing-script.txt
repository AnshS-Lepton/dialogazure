 WITH duplicates AS (
  SELECT
    ctid,  -- unique identifier for each row
    ROW_NUMBER() OVER (
      PARTITION BY layer_id, db_column_name
      ORDER BY ctid
    ) AS rn
  FROM data_uploader_template
  WHERE layer_id = (select layer_id  from layer_details ld where layer_name='Cable') 
  )
	DELETE FROM data_uploader_template
	WHERE ctid IN (
	  SELECT ctid FROM duplicates WHERE rn > 1
    );
